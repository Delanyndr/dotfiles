# Default configuration for syslinux, for BIOS systems

Ensure root partition is set with legacy boot flag

Install syslinux

run 'syslinux-install_update -i -a -m'

create prefered /boot/syslinux/syslinux.cfg

EXAMPLE syslinux.cfg

PROMPT 0
TIMEOUT 50
DEFAULT arch

LABEL arch
    LINUX ../vmlinux-linux
    APPEND root=/dev/<root> rw
    INITRD ../intel-ucode.img,../initramfs-linux.img
